ownership

      fn main() {
          let x = Box::new(5);
          add_one(x);
          println!("{}", x);
      }

      fn add_one(mut num: Box) {
          *num += 1;
      }
box allocates memory on the heap

borrowing

      fn main() {
          let mut x = 5;
          add_one(&mut x);
          println!("{}", x);
      }
      
      fn add_one(num: &mut i32) {
          *num += 1;
      }
      

        has temporary ownership
